DROP DATABASE STUDENTS
CREATE DATABASE STUDENTS
USE STUDENTS
CREATE TABLE Enseignant(idEnseignant INT NOT NULL AUTO_INCREMENT,prenom VARCHAR(50),nom VARCHAR(50),nbHeure DECIMAL(15,2),PRIMARY KEY(idEnseignant))
CREATE TABLE Salle(idSalle INT NOT NULL AUTO_INCREMENT,capacite INT,dispo BOOLEAN,typeSalle VARCHAR(5),PRIMARY KEY(idSalle))
CREATE TABLE Matiere(idMatiere INT NOT NULL AUTO_INCREMENT,nom VARCHAR(50),PRIMARY KEY(idMatiere))
CREATE TABLE Suitcours(idSuit INT NOT NULL AUTO_INCREMENT,nomGroupe VARCHAR(5),taille INT,typeGroupe VARCHAR(5),PRIMARY KEY(idSuit))
CREATE TABLE Etudiant(idEtudiant INT NOT NULL AUTO_INCREMENT,nom VARCHAR(50),prenom VARCHAR(50),parcours VARCHAR(5),PRIMARY KEY(idEtudiant))
CREATE TABLE Enseigne(idEnseignant INT,idMatiere INT,PRIMARY KEY(idEnseignant, idMatiere),FOREIGN KEY(idEnseignant) REFERENCES Enseignant(idEnseignant),FOREIGN KEY(idMatiere) REFERENCES Matiere(idMatiere))
CREATE TABLE Cours(idEnseignant INT,idSalle INT,idMatiere INT,idSuit INT,duree DOUBLE(2,1),journee DATE,horaire INT,PRIMARY KEY(idEnseignant, idSalle, idMatiere, idSuit),FOREIGN KEY(idEnseignant) REFERENCES Enseignant(idEnseignant),FOREIGN KEY(idSalle) REFERENCES Salle(idSalle),FOREIGN KEY(idMatiere) REFERENCES Matiere(idMatiere),FOREIGN KEY(idSuit) REFERENCES SuitCours(idSuit))
